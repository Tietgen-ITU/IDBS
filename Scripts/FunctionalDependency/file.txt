SELECT 'Rentals: PID --> HID' AS FD,
CASE WHEN COUNT(*)=0 THEN 'MAY HOLD'
ELSE 'does not hold' END AS VALIDITY
FROM (
SELECT P.PID
FROM Rentals P
GROUP BY P.PID
HAVING COUNT(DISTINCT P.HID) > 1
) X;
